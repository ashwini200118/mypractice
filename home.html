Pushing this folder to GitHub

# in d:\html progms
git init

# create a GitHub repo (web or use gh CLI)
# then add remote (replace USERNAME/REPO_NAME)

# HTTPS
git remote add origin https://github.com/USERNAME/REPO_NAME.git

# or SSH
git remote add origin git@github.com:USERNAME/REPO_NAME.git

git branch -M main
git push -u origin main

# subsequent updates:
git add .
git commit -m "Update"
git push
